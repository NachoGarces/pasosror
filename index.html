#Crear

Crear proyecto

rails new plantilla -d postgresql

Entrar al proyecto

cd plantilla

Configurar claves de db

config/database.yml {

  username:
  password:

}

Crear base de datos

rails db:create

#Bootstrap

Agregar gemas

bundle add bootstrap
bundle add jquery-rails
bundle add popper

Configurar SCSS

app/assets/stylesheets/appliction.'S'css {

  @import "bootstrap";

}

Configurar JS

app/javascript/application.js {

  //= require jquery3
  //= require popper
  //= require bootstrap-sprockets
  import "@hotwired/turbo-rails"
  import "controllers"

}

Initializers

config/initializers/assets.rb {

  Rails.application.config.assets.precompile += w( application.scss bootstrap.min.js popper.js )

}

Importmap

config/importmap.rb {

  pin "popper", to: 'popper.js', preload: true
  pin "bootstrap", to: 'bootstrap.min.js', preload: true

}

#Navbar y footer

Plantillas

app/views/"shared/_navbar.html.erb" {

  "Agregar el bootstrap de preferencia"

}

app/views/"shared/_footer.html.erb" {

  "Agregar el bootstrap de preferencia"

}

Render a plantillas

app/views/layouts/application.html.erb {

  <%= render "shared/navbar" %>

  <div class='container'>
   <%= yield %>
  </div>

  <%= render "shared/footer" %>

}

#Vista inicial

Generar controlador

rails g controller welcome index

Welcome como vista por defecto

config/routes.rb {

  root "welcome#index"

}

#Devise

Agregar devise

bundle add devise

Instalar devise en el proyecto

rails g devise:install

Generar devise USER

rails g devise user

Correr la migracion creada

rails db:migrate

Para ver y modificar devise se generan las vistas

rails g devise:views

Generar controlador devise

rails g devise:controllers users
